// ─── Contact wizard (Phase 4A) ──────────────────────────────────────
document.querySelectorAll("[data-wizard]").forEach((wizard) => {
  const steps = Array.from(wizard.querySelectorAll("[data-wizard-step]"));
  const laneTiles = Array.from(wizard.querySelectorAll(".lane-tile"));
  const backButtons = Array.from(wizard.querySelectorAll("[data-wizard-back]"));
  const progressFill = wizard.querySelector(".wizard-progress__bar-fill");
  const stepLabels = Array.from(wizard.querySelectorAll("[data-wizard-step-label]"));
  let currentLane = null;

  const showStep = (stepId) => {
    steps.forEach((s) => {
      const match = s.getAttribute("data-wizard-step") === stepId &&
        (!s.hasAttribute("data-wizard-lane") || s.getAttribute("data-wizard-lane") === currentLane);
      s.setAttribute("data-active", String(match));
    });
    if (progressFill) {
      progressFill.style.width = stepId === "1" ? "30%" : stepId === "success" ? "100%" : "70%";
    }
    stepLabels.forEach((label) => {
      const num = label.getAttribute("data-wizard-step-label");
      label.setAttribute("data-current", String(num === stepId));
      label.setAttribute("data-done", String(Number(num) < Number(stepId)));
    });

    tile.addEventListener("click", () => {

    });

      showStep("1");

**JS module** (append to `site.js`):

```javascript
// ─── Hero self-segmentation switch (Phase 5A) ───────────────────────
const segmentSwitch = document.querySelector(".segment-switch");
if (segmentSwitch) {
  const buttons = Array.from(segmentSwitch.querySelectorAll("[data-segment]"));
  const panels = Array.from(document.querySelectorAll("[data-segment-panel]"));
  const STORAGE_KEY = "nas-segment";

  const activate = (segment) => {
    buttons.forEach((btn) => {
      btn.setAttribute("aria-selected", String(btn.getAttribute("data-segment") === segment));
    });
    panels.forEach((panel) => {
      panel.setAttribute("data-active", String(panel.getAttribute("data-segment-panel") === segment));
    });
    try { localStorage.setItem(STORAGE_KEY, segment); } catch { /* no-op */ }
    track("segment_switch", { segment });
  };

  buttons.forEach((btn) => {
    btn.addEventListener("click", () => {
      activate(btn.getAttribute("data-segment"));
    });
  });

  // Keyboard navigation
  segmentSwitch.addEventListener("keydown", (e) => {
    const idx = buttons.findIndex((b) => b === document.activeElement);
    if (idx < 0) return;
    if (e.key === "ArrowRight" || e.key === "ArrowDown") {
      e.preventDefault();
      const next = buttons[(idx + 1) % buttons.length];
      next.focus();
      activate(next.getAttribute("data-segment"));
    } else if (e.key === "ArrowLeft" || e.key === "ArrowUp") {
      e.preventDefault();
      const prev = buttons[(idx - 1 + buttons.length) % buttons.length];
      prev.focus();
      activate(prev.getAttribute("data-segment"));
    }
  });

  // Restore from localStorage or query param
  const urlParams = new URLSearchParams(window.location.search);
  const paramSegment = urlParams.get("segment");
  const storedSegment = (() => { try { return localStorage.getItem(STORAGE_KEY); } catch { return null; } })();
  const initial = paramSegment || storedSegment;
  if (initial && buttons.some((b) => b.getAttribute("data-segment") === initial)) {
    activate(initial);
  }
}
```

// ─── System map micro-simulator (Phase 6C) ──────────────────────────
document.querySelectorAll(".system-map").forEach((map) => {
  const steps = Array.from(map.querySelectorAll(".system-map__step"));
  const progressFill = map.querySelector(".system-map__progress-fill");
  const detail = map.querySelector(".system-map__detail");
  const details = JSON.parse(map.getAttribute("data-step-details") || "[]");

  steps.forEach((step, i) => {
    step.addEventListener("click", () => {
      steps.forEach((s) => s.setAttribute("data-active", "false"));
      step.setAttribute("data-active", "true");
      const progress = 30 + ((i + 1) / steps.length) * 70;
      if (progressFill) progressFill.style.width = `${progress}%`;
      if (detail && details[i]) detail.textContent = details[i];
      track("system_map_step", { step: i, name: step.querySelector(".system-map__step-name")?.textContent });
    });
  });

  if (steps[0]) steps[0].setAttribute("data-active", "true");
});
